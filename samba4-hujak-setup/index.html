Заметка будущему себе.

Понадобилось срочно зашарить файлов соседу, пришлось поднимать Самбу4 в режиме интенсивного hujak-hujak. Итоговый конфиг для шар с гостевым доступом:

[email protected]:~#  cat /etc/samba/smb.conf | grep -v '^#' | grep -v '^;' | grep -v '^$'
[global]
   workgroup = WORKGROUP
   server string = My smb share
   dns proxy = yes
   interfaces = 10.0.0.0/8 enp1s0
   guest account = smb
   log file = /var/log/samba/log.%m
   max log size = 1000
   syslog = 0
   security = user
   map to guest = Bad Password
   load printers = no
   acl allow execute always = True

[share]
    comment = "Some files"
    path = /mnt/data/files
    browseable = yes
    read only = no
    guest ok = yes
    guest account = smb
    create mask = 0666
    directory mask = 0775
