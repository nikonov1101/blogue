Вывод командной строки в Golang

8 августа 2016 г. 13:33

Понадобилось работать с многострочным выводом shell-команды из Go, получился такой код:

package main

import (
    "bufio"
    "fmt"
    "log"
    "os/exec"
    "strings"
)

func main() {
    // выполняем команду и получаем вывод в виде массива байтов
    out, err := exec.Command("ls", "-l").Output()
    if err != nil {
        log.Fatal(err)
    }

    // конвертируем вывод в строку
    cmdString := fmt.Sprintf("%s", out)

    // инициализируем ридер строк из потокового буфера
    scanner := bufio.NewScanner(strings.NewReader(cmdString))

    // читаем вывод построчно
    for scanner.Scan() {
        line := scanner.Text()
        // делаем с выводом все, что угодно
        fmt.Println(line)
    }
}

P.S.: Сам код и эта записть написаны из машины где-то в Купчино.
